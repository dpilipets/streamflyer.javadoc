017    package com.googlecode.streamflyer.experimental.range;
018    
019    import com.googlecode.streamflyer.core.AfterModification;
020    import com.googlecode.streamflyer.core.Modifier;
021    import com.googlecode.streamflyer.experimental.stateful.State;
022    import com.googlecode.streamflyer.experimental.stateful.StatefulModifier;
023    import com.googlecode.streamflyer.internal.thirdparty.ZzzValidate;
024    import com.googlecode.streamflyer.regex.RegexModifier;
025    
026    /**
027     * This modifier deletes all data from the character stream that is not placed
028     * between the given start and end tag. This is a bit similar to Perl's or
029     * Ruby's range operator.
030     * <p>
031     * Additionally, you can choose whether the start or end tags shall be removed
032     * from the input as well.
033     * <p>
034     * EXAMPLES: <code><pre>
035      start tag   end tag     init. input                modified input
036      ----------- ----------- ----- -------------------- --------------
037      STA (excl.) END (excl.) END   "from STA to END."   " to " 
038      STA (incl.) END (incl.) END   "from STA to END."   "STA to END"
039      STA (incl.) END (incl.) STA   "from STA to END."   "from STA to END"
040      S (excl.)   E (excl.)   E     "a S, E, S, E, ..."  ", , "  
041      S (incl.)   E (incl.)   E     "a S, E, S, E, ..."  "S, ES, E" 
042      
043      </pre></code>
044     * 
045     * @author rwoo
046     * 
047     * @since 13.09.2011
048     */
049    public class RangeFilterModifier implements Modifier {
050    
051        private StatefulModifier statefulModifier;
052    
053        /**
054         * The match processor of both given modifiers are replaced with another
055         * one. So you don't have to care about the match processor you have used on
056         * 
057         * @param startModifier
058         *            Must not be <code>null</code>.
059         * @param endModifier
060         *            Must not be <code>null</code>.
061         * @param includeStart
062         *            True if the start tag shall not be deleted from the input.
063         *            False if the start tag shall be deleted from the input.
064         * @param includeEnd
065         *            True if the end tag shall not be deleted from the input. False
066         *            if the end tag shall be deleted from the input.
067         * @param initiallyBeforeEnd
068         *            True if the given input shall be parsed assuming the input is
069         *            prefixed with an end tag. False if the given input shall be
070         *            parsed assuming the input is prefixed with a start tag.
071         */
072        public RangeFilterModifier(RegexModifier startModifier,
073                RegexModifier endModifier, boolean includeStart,
074                boolean includeEnd, boolean initiallyBeforeEnd) {
075    
076            ZzzValidate.notNull(startModifier, "startModifier must not be null");
077            ZzzValidate.notNull(endModifier, "endModifier must not be null");
078    
079            // *** create two state that will be used by the stateful modifier
080    
081            // create the state for the end tag
082            RangeFilterState beforeEndState = new RangeFilterState(endModifier,
083                    false, !includeEnd);
084            endModifier.setMatchProcessor(beforeEndState);
085    
086            // create the state for the start tag
087            RangeFilterState beforeStartState = new RangeFilterState(startModifier,
088                    true, !includeStart);
089            startModifier.setMatchProcessor(beforeStartState);
090    
091            // *** link the states
092            beforeEndState.setNextState(beforeStartState);
093            beforeStartState.setNextState(beforeEndState);
094    
095            // define the initial state
096            State initialState;
097            if (initiallyBeforeEnd) {
098                initialState = beforeEndState;
099            } else {
100                initialState = beforeStartState;
101            }
102    
103            this.statefulModifier = new StatefulModifier(initialState);
104        }
105    
106        /**
107         * @see com.googlecode.streamflyer.core.Modifier#modify(java.lang.StringBuilder,
108         *      int, boolean)
109         */
110        @Override
111        public AfterModification modify(StringBuilder characterBuffer,
112                int firstModifiableCharacterInBuffer, boolean endOfStreamHit) {
113    
114            // delegate
115            return statefulModifier.modify(characterBuffer,
116                    firstModifiableCharacterInBuffer, endOfStreamHit);
117        }
118    }
