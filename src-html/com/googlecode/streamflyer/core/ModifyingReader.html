019import java.io.BufferedReader;
020import java.io.IOException;
021import java.io.Reader;
022import java.util.ArrayList;

029/**
030 * A {@link Reader} that allows a {@link Modifier} to modify the characters in
031 * the underlying stream.
032 * <p>
033 * This class is not synchronized.
034 * <p>
035 * {@link #markSupported() Marking} is not supported by this reader.
036 * 
037 * @author rwoo
038 * @since 06.05.2011
039 */
040public class ModifyingReader extends Reader {
041
042    //
043    // injected properties
044    //
045
046    /**
047     * The reader that the reads from the underlying character stream.
048     */
049    protected Reader delegate;
050
051    /**
052     * The modifier thats replaces, deletes, inserts characters of the
053     * underlying stream, and manages the buffer size.
054     */
055    private Modifier modifier;
056
057
058    //
059    // properties that represent the mutable state
060    //
061
062    /**
063     * At its begin the buffer contains unmodifiable characters (might be needed
064     * for look-behind matches as known from regular expression matching). After
065     * the unmodifiable characters the buffer holds the modifiable characters.
066     * See {@link #firstModifiableCharacterInBuffer}.
067     */
068    private StringBuilder characterBuffer;
069
070
071    /**
072     * The position of the first character in the {@link #characterBuffer input
073     * buffer} that is modifiable.
074     * <p>
075     * This character and the following characters in {@link #characterBuffer}
076     * can be modified by the {@link #modifier}.
077     */
078    private int firstModifiableCharacterInBuffer;
079
080
081    /**
082     * The value is taken from
083     * {@link AfterModification#getNewMinimumLengthOfLookBehind()} after
084     * {@link Modifier#modify(StringBuilder, int, boolean)} is called. The value
085     * is <em>requested</em> by the {@link Modifier}.
086     * <p>
087     * If this value is greater than {@link #firstModifiableCharacterInBuffer},
088     * the modifier is probably faulty.
089     */
090    private int minimumLengthOfLookBehind;
091
092    /**
093     * The value is the sum of
094     * {@link AfterModification#getNewMinimumLengthOfLookBehind()} and
095     * {@link AfterModification#getNewNumberOfChars()} after
096     * {@link Modifier#modify(StringBuilder, int, boolean)} is called. The value
097     * is <em>requested</em> by the {@link Modifier}.
098     */
099    private int requestedNumCharactersInBuffer;
100
101    /**
102     * The value is initially taken from
103     * {@link AfterModification#getNumberOfCharactersToSkip()} after
104     * {@link Modifier#modify(StringBuilder, int, boolean)} is called. Then
105     * during reading characters from the reader this value is decremented.
106     */
107    private int numberOfCharactersToSkip;
108
109    /**
110     * True if the end of stream was detected.
111     */
112    private boolean endOfStreamHit = false;
113
114    /**
115     * The holds the last {@link AfterModification} provided by the
116     * {@link #modifier}. This property serves debugging purposes only.
117     */
118    private AfterModification lastAfterModificationForDebuggingOnly = null;
119
120    /**
121     * @param reader The underlying reader that provides the original, not
122     *        modified characters. For optimal performance choose a
123     *        {@link BufferedReader}.
124     * @param modifier the object that modifies the stream.
125     */
126    public ModifyingReader(Reader reader, Modifier modifier) {
127        super();
128
129        this.delegate = reader;
130
131        // take injected properties
132        this.modifier = modifier;
133
134        // initialize the mutable state
135        this.numberOfCharactersToSkip = 0;
136        this.minimumLengthOfLookBehind = 0; // we cannot do anything else
137        this.firstModifiableCharacterInBuffer = 0;
138        this.requestedNumCharactersInBuffer = 1; // any value
139        this.characterBuffer = new StringBuilder(requestedNumCharactersInBuffer);
140        adjustCapacityOfBuffer();
141    }
142
143    /**
144     * Updates the input buffer according to {@link #minimumLengthOfLookBehind}
145     * and {@link #requestedNumCharactersInBuffer}, and then fills the buffer up
146     * to its capacity.
147     */
148    private void updateBuffer() throws IOException {
149
150        removeCharactersInBufferNotNeededAnyLonger();
151
152        adjustCapacityOfBuffer();
153
154        fill();
155    }
156
157    /**
158     * Deletes those characters at the start of the buffer we do not need any
159     * longer.
160     */
161    private void removeCharactersInBufferNotNeededAnyLonger() {
162
163        int charactersToDelete = firstModifiableCharacterInBuffer
164                - minimumLengthOfLookBehind;
165
166        if (charactersToDelete > 0) {
167            characterBuffer.delete(0, charactersToDelete);
168            firstModifiableCharacterInBuffer -= charactersToDelete;
169        }
170        else if (charactersToDelete < 0) {
171            onFaultyModifier(-52, charactersToDelete
172                    + " characters to delete but this is not possible ("
173                    + firstModifiableCharacterInBuffer + ","
174                    + minimumLengthOfLookBehind + ")");
175        }
176    }
177
178    /**
179     * Adjusts the capacity of the buffer.
180     */
181    protected void adjustCapacityOfBuffer() {
182
183        // is the current capacity not big enough?
184        if (characterBuffer.capacity() < requestedNumCharactersInBuffer) {
185
186            // increase the capacity (we delegate to the default behavior of
187            // StringBuffer)
188            characterBuffer.ensureCapacity(requestedNumCharactersInBuffer);
189
190        }
191
192    }
193
194    /**
195     * Reads more characters into the input buffer. This method will block until
196     * the buffer is filled with {@link #requestedNumCharactersInBuffer}
197     * characters , or an I/O error occurs, or the end of the stream is reached.
198     */
199    private void fill() throws IOException {
200
201        int length = requestedNumCharactersInBuffer - characterBuffer.length();
202        if (length <= 0) {
203            // nothing to do
204            return;
205        }
206
207
208        char[] buffer = new char[length];
209        int offset = 0;
210
211        while (length > 0) {
212
213            int readChars = delegate.read(buffer, offset, length);
214            if (readChars != -1) {
215                characterBuffer.append(buffer, offset, readChars);
216                offset += readChars;
217                length -= readChars;
218            }
219            else {
220                endOfStreamHit = true;
221                break;
222            }
223
224        }
225
226    }
227
228    protected void onFaultyModifier(int errorCode, String errorMessage) {
229        // should we silently ignore any errors and fallback to a meaningful
230        // behavior? No because a faulty modifier should be fixed by a
231        // ValidModifier, i.e. the user must take the responsibility for the
232        // errors and deal with them. Therefore, we throw an exception here.
233
234        // (1) collect the current state of the involved objects
235        Map<String, Object> description = new HashMap<String, Object>();
236
237        // describe arguments
238        description.put("argument errorCode", errorCode);
239        description.put("argument errorMessage", errorMessage);
240
241        // describe reader
242        description.put("this", this);
243
244        // (2) create an error message
245        List<String> keys = new ArrayList<String>(description.keySet());
246        Collections.sort(keys);
247        StringBuilder sb = new StringBuilder();
248        for (String key : keys) {
249            sb.append("\n" + key + ": " + description.get(key));
250        }
251
252        // (3) throw exception
253        throw new FaultyModifierException(sb.toString(), description);
254    }

256    //
257    // interface Reader
258    //
259
260    /**
261     * TODO more doc
262     * 
263     * @see java.io.Reader#read(char[], int, int)
264     */
265    @Override
266    public int read(char[] cbuf, int off, int len) throws IOException {
267        int index = 0;
268        int read = 0;
269        while (index < len && (read = readCharacter()) != -1) {
270            cbuf[off + index] = (char) read;
271            index++;
272        }
273
274        if (index == 0 && read == -1) {
275            return -1;
276        }
277        else {
278            return index;
279        }
280    }
281
282    /**
283     * Contract is similar to {@link java.io.Reader#read()}.
284     * <p>
285     * TODO more doc
286     */
287    protected int readCharacter() throws IOException {
288
289        if (numberOfCharactersToSkip == 0) {
290
291            boolean modifyAgainImmediately = false;
292
293            AfterModification afterModification = null;
294            do {
295
296                updateBuffer();
297
298                afterModification = modifier.modify(characterBuffer,
299                        firstModifiableCharacterInBuffer, endOfStreamHit);
300
301                lastAfterModificationForDebuggingOnly = afterModification;
302
303                // update minimumLengthOfLookBehind
304                minimumLengthOfLookBehind = afterModification
305                        .getNewMinimumLengthOfLookBehind();
306                if (minimumLengthOfLookBehind > firstModifiableCharacterInBuffer
307                        + afterModification.getNumberOfCharactersToSkip()) {
308                    onFaultyModifier(-11, "Requested Look Behind"
309                            + " is impossible because there are not enough "
310                            + "characters in the stream.");
311                }
312
313                // update requestedNumCharactersInBuffer
314                requestedNumCharactersInBuffer = minimumLengthOfLookBehind
315                        + afterModification.getNewNumberOfChars();
316                if (requestedNumCharactersInBuffer < minimumLengthOfLookBehind + 1) {
317                    onFaultyModifier(-13, "Requested Capacity"
318                            + " is two small because there must at least one"
319                            + " unread characters available after the "
320                            + "look behind characters characters in the"
321                            + " stream.");
322                }
323
324                modifyAgainImmediately = false;
325                // do we have to read at least a single character as there is no
326                // modifiable character left in the character buffer?
327                if (firstModifiableCharacterInBuffer >= characterBuffer
328                        .length() && !endOfStreamHit) {
329                    // yes, we need fresh input ->
330                    modifyAgainImmediately = true;
331                }
332                // has the modifier requested a modification at the current
333                // position?
334                else if (afterModification.isModifyAgainImmediately()) {
335                    modifyAgainImmediately = true;
336                }
337
338            } while (modifyAgainImmediately);
339
340            numberOfCharactersToSkip = afterModification
341                    .getNumberOfCharactersToSkip();
342
343            if (!afterModification.isModifyAgainImmediately()
344                    && numberOfCharactersToSkip == 0 && !endOfStreamHit) {
345                onFaultyModifier(-16, "Not a single characters shall be "
346                        + "skipped but this is not possible of "
347                        + "modifyAgain() returns false and the end of "
348                        + "stream is not reached yet.");
349            }
350            numberOfCharactersToSkip--;
351        }
352        else {
353            numberOfCharactersToSkip--;
354        }
355
356        // is the end of the stream reached?
357        if (firstModifiableCharacterInBuffer >= characterBuffer.length()) {
358            // yes -> return -1
359            return -1;
360        }
361        else {
362            // no -> return the next unread character
363            char result = characterBuffer
364                    .charAt(firstModifiableCharacterInBuffer);
365
366            firstModifiableCharacterInBuffer++;
367
368            return result;
369        }
370    }

376    /**
377     * @see java.lang.Object#toString()
378     */
379    @Override
380    public String toString() {
381        StringBuilder builder = new StringBuilder();
382        builder.append("ModifyingReader [\ncharacterBuffer=");
383        builder.append(characterBuffer);
384        builder.append(", \nnextUnreadCharacterInBuffer=");
385        builder.append(firstModifiableCharacterInBuffer);
386        builder.append(", \nmodifier=");
387        builder.append(modifier);
388        builder.append(", \nminimumLengthOfLookBehind=");
389        builder.append(minimumLengthOfLookBehind);
390        builder.append(", \nrequestedCapacityOfCharacterBuffer=");
391        builder.append(requestedNumCharactersInBuffer);
392        builder.append(", \nlockedCharacters=");
393        builder.append(numberOfCharactersToSkip);
394        builder.append(", \nendOfStreamHit=");
395        builder.append(endOfStreamHit);
396        builder.append(", \nlastModificationForDebuggingOnly=");
397        builder.append(lastAfterModificationForDebuggingOnly);
398        builder.append("]");
399        return builder.toString();
400    }
