018
019import java.io.BufferedWriter;
020import java.io.IOException;
021import java.io.Writer;
022import java.util.ArrayList;
023import java.util.Collections;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;

029/**
030 * A {@link Writer} that allows a {@link Modifier} to modify the characters in
031 * the underlying stream.
032 * <p>
033 * This class is not synchronized.
034 * <p>
035 * ATTENTION! This writer flushes only characters that are confirmed by the
036 * given {@link Modifier}. EXAMPLE: Assume you wrote 50 bytes to the writer and
037 * the modifier confirmed 25 of them by
038 * {@link AfterModification#getNumberOfCharactersToSkip() skipping } them, then
039 * subsequent flushing will write only these 25 confirmed characters.
040 * 
041 * @author rwoo
042 * @since 06.05.2011
043 */
044public class ModifyingWriter extends Writer {
045
046    //
047    // injected properties
048    //
049
050    protected Writer delegate;
051
052    /**
053     * The modifier thats replaces, deletes, inserts characters of the
054     * underlying stream, and manages the buffer size.
055     */
056    private Modifier modifier;
057
058
059    //
060    // properties that represent the mutable state
061    //
062
063    /**
064     * At its begin the buffer contains unmodifiable characters (might be needed
065     * for look-behind matches as known from regular expression matching). After
066     * the unmodifiable characters the buffer holds the modifiable characters.
067     * See {@link #firstModifiableCharacterInBuffer}.
068     */
069    private StringBuilder characterBuffer;
070
071
072    /**
073     * The position of the first character in the {@link #characterBuffer input
074     * buffer} that is modifiable.
075     * <p>
076     * This character and the following characters in {@link #characterBuffer}
077     * can be modified by the {@link #modifier}.
078     */
079    private int firstModifiableCharacterInBuffer;
080
081
082    /**
083     * The value is taken from
084     * {@link AfterModification#getNewMinimumLengthOfLookBehind()} after
085     * {@link Modifier#modify(StringBuilder, int, boolean)} is called. The value
086     * is <em>requested</em> by the {@link Modifier}.
087     * <p>
088     * If this value is greater than {@link #firstModifiableCharacterInBuffer},
089     * the modifier is probably faulty.
090     */
091    private int minimumLengthOfLookBehind;
092
093    /**
094     * The value is the sum of
095     * {@link AfterModification#getNewMinimumLengthOfLookBehind()} and
096     * {@link AfterModification#getNewNumberOfChars()} after
097     * {@link Modifier#modify(StringBuilder, int, boolean)} is called. The value
098     * is <em>requested</em> by the {@link Modifier}.
099     */
100    private int requestedNumCharactersInBuffer;
101
102    /**
103     * The value is initially taken from
104     * {@link AfterModification#getNumberOfCharactersToSkip()} after
105     * {@link Modifier#modify(StringBuilder, int, boolean)} is called. Then
106     * during reading characters from the reader this value is decremented.
107     */
108    private int numberOfCharactersToSkip;
109
110    /**
111     * True if the end of stream was detected.
112     */
113    private boolean endOfStreamHit = false;
114
115    /**
116     * The holds the last modification provided by the {@link #modifier}. This
117     * property serves debugging purposes only.
118     */
119    private AfterModification lastAfterModificationForDebuggingOnly = null;
120
121    /**
122     * @param writer The underlying writer that provides the original, not
123     *        modified characters. For optimal performance choose a
124     *        {@link BufferedWriter}.
125     * @param modifier
126     */
127    public ModifyingWriter(Writer writer, Modifier modifier) {
128        super(writer);
129
130        // take injected properties
131        this.modifier = modifier;
132        this.delegate = writer;
133
134        // initialize the mutable state
135        this.numberOfCharactersToSkip = 0;
136        this.minimumLengthOfLookBehind = 0; // we cannot do anything else
137        this.firstModifiableCharacterInBuffer = 0;
138        this.requestedNumCharactersInBuffer = 1; // any value
139        this.characterBuffer = new StringBuilder(requestedNumCharactersInBuffer);
140        // adjustCapacityOfBuffer();
141    }
142
143    //
144    // interface Writer
145    //
146
147    /**
148     * @see java.io.Writer#flush()
149     */
150    @Override
151    public void flush() throws IOException {
152
153        // delegate to the underlying writer
154        delegate.flush();
155    }
156
157    /**
158     * @see java.io.Writer#close()
159     */
160    @Override
161    public void close() throws IOException {
162
163        endOfStreamHit = true;
164
165        // modify as long as characters are skipped (modify again makes not
166        // much sense but there may some use cases that require modfiyAgain even
167        // on the position after the last character)
168        while (modify()) {
169            // loop
170        }
171
172        // write the remaining bytes to the underlying writer
173        delegate.append(characterBuffer, firstModifiableCharacterInBuffer,
174                characterBuffer.length());
175
176        // clear same variables (only to tidy up)
177        characterBuffer = null;
178
179        flush();
180    }
181
182    /**
183     * @see java.io.Writer#write(char[], int, int)
184     * @param off the first character in cbuf not appended to characterBuffer
185     *        yet
186     * @param len the number of characters in cbuf not appended to //
187     *        characterBuffer yet
188     */
189    @Override
190    public void write(char[] cbuf, int off, int len) throws IOException {
191
192        // is the stream already closed?
193        if (endOfStreamHit) {
194            throw new IOException("the stream is already closed");
195        }
196
197        // are there some characters to append to the characterBuffer yet?
198        while (len > 0) {
199            // yes -> append characters
200
201            // determine the number of characters to add to the buffer
202            int numberOfCharactersToAppend;
203            // are there are more characters available than requested?
204            if (len > requestedNumCharactersInBuffer - characterBuffer.length()) {
205                // yes -> we won't append more characters than are requested (in
206                // order to save memory)
207                numberOfCharactersToAppend = requestedNumCharactersInBuffer
208                        - characterBuffer.length();
209            }
210            else {
211                // we will append all given characters
212                numberOfCharactersToAppend = len;
213            }
214
215            // append the characters to the characterBuffer
216            characterBuffer.append(cbuf, off, numberOfCharactersToAppend);
217
218            // update the variables offset and length
219            off = off + numberOfCharactersToAppend;
220            len = len - numberOfCharactersToAppend;
221
222            // modify if there are enough characters in the buffer
223            while (characterBuffer.length() >= requestedNumCharactersInBuffer
224                    && modify()) {
225                // loop
226            }
227
228        }
229    }
230
231    /**
232     * @return Returns true if some characters are skipped, i.e. written to the
233     *         underlying writer, or the content of the buffer shall be modified
234     *         again immediately.
235     * @throws IOException
236     */
237    private boolean modify() throws IOException {
238        AfterModification afterModification = modifier.modify(characterBuffer,
239                firstModifiableCharacterInBuffer, endOfStreamHit);
240
241        lastAfterModificationForDebuggingOnly = afterModification;
242
243        numberOfCharactersToSkip = afterModification
244                .getNumberOfCharactersToSkip();
245
246        boolean someCharactersSkippedOrModifyAgainImmediately = false;
247
248        if (afterModification.isModifyAgainImmediately()) {
249
250            if (endOfStreamHit) {
251                // A modifier might do "modify again" although there is no
252                // more input. Assume the modifier want to edit its own
253                // insertions -> Therefore, we throw no exception here.
254            }
255
256            // nothing to do apart from adjusting capacity and look-behind
257
258            someCharactersSkippedOrModifyAgainImmediately = true;
259
260        }
261        else {
262
263            // write away the locked characters
264            if (numberOfCharactersToSkip > 0) {
265                int end = firstModifiableCharacterInBuffer
266                        + numberOfCharactersToSkip;
267                if (end > characterBuffer.length()) {
268                    onFaultyModifier(-51, String.format("You try to skip "
269                            + "characters that you have not"
270                            + " seen yet(%s %s %s %s)",
271                            firstModifiableCharacterInBuffer,
272                            numberOfCharactersToSkip, characterBuffer.length(),
273                            characterBuffer));
274                }
275
276                delegate.append(characterBuffer,
277                        firstModifiableCharacterInBuffer, end);
278                someCharactersSkippedOrModifyAgainImmediately = true;
279                firstModifiableCharacterInBuffer = end;
280            }
281            else { // if (numberOfCharactersToSkip == 0)
282
283                // This block is usually entered when the modifier decided to do
284                // nothing at the position after the last characters of the
285                // stream
286
287            }
288        }
289
290
291        // update minimumLengthOfLookBehind
292        minimumLengthOfLookBehind = afterModification
293                .getNewMinimumLengthOfLookBehind();
294        if (minimumLengthOfLookBehind > firstModifiableCharacterInBuffer) {
295            onFaultyModifier(-11, "Requested Look Behind"
296                    + " is impossible because there are not enough "
297                    + "characters in the stream.");
298        }
299
300        // update requestedNumCharactersInBuffer
301        requestedNumCharactersInBuffer = minimumLengthOfLookBehind
302                + afterModification.getNewNumberOfChars();
303        if (requestedNumCharactersInBuffer < minimumLengthOfLookBehind + 1) {
304            onFaultyModifier(-13, "Requested Capacity"
305                    + " is two small because there must at least one"
306                    + " unread characters available after the "
307                    + "look behind characters characters in the" + " stream.");
308        }
309
310        updateBuffer();
311
312        return someCharactersSkippedOrModifyAgainImmediately;
313    }
314
315    /**
316     * Updates the input buffer according to {@link #minimumLengthOfLookBehind}
317     * and {@link #requestedNumCharactersInBuffer}, and then fills the buffer up
318     * to its capacity.
319     */
320    private void updateBuffer() {
321
322        removeCharactersInBufferNotNeededAnyLonger();
323
324        adjustCapacityOfBuffer();
325
326    }
327
328    /**
329     * Deletes those characters at the start of the buffer we do not need any
330     * longer.
331     */
332    private void removeCharactersInBufferNotNeededAnyLonger() {
333
334        int charactersToDelete = firstModifiableCharacterInBuffer
335                - minimumLengthOfLookBehind;
336
337        if (charactersToDelete > 0) {
338            characterBuffer.delete(0, charactersToDelete);
339            firstModifiableCharacterInBuffer -= charactersToDelete;
340        }
341        else if (charactersToDelete < 0) {
342            onFaultyModifier(-52, charactersToDelete
343                    + " characters to delete but this is not possible ("
344                    + firstModifiableCharacterInBuffer + ","
345                    + minimumLengthOfLookBehind + ")");
346        }
347    }
348
349    /**
350     * Adjusts the capacity of the buffer.
351     */
352    private void adjustCapacityOfBuffer() {
353
354        // is the current capacity not big enough?
355        if (characterBuffer.capacity() < requestedNumCharactersInBuffer) {
356
357            // increase the capacity (we delegate to the default behavior of
358            // StringBuffer)
359            characterBuffer.ensureCapacity(requestedNumCharactersInBuffer);
360
361        }
362
363    }
364
365
366    protected void onFaultyModifier(int errorCode, String errorMessage) {
367        // should we silently ignore any errors and fallback to a meaningful
368        // behavior? No because a faulty modifier should be fixed by a
369        // ValidModifier, i.e. the user must take the responsibility for the
370        // errors and deal with them. Therefore, we throw an exception here.
371
372        // (1) collect the current state of the involved objects
373        Map<String, Object> description = new HashMap<String, Object>();
374
375        // describe arguments
376        description.put("argument errorCode", errorCode);
377        description.put("argument errorMessage", errorMessage);
378
379        // describe reader
380        description.put("this", this);
381
382        // (2) create an error message
383        List<String> keys = new ArrayList<String>(description.keySet());
384        Collections.sort(keys);
385        StringBuilder sb = new StringBuilder();
386        for (String key : keys) {
387            sb.append("\n" + key + ": " + description.get(key));
388        }
389
390        // (3) throw exception
391        throw new FaultyModifierException(sb.toString(), description);
392    }

394    /**
395     * @see java.lang.Object#toString()
396     */
397    @Override
398    public String toString() {
399        StringBuilder builder = new StringBuilder();
400        builder.append("ModifyingWriter [\ndelegate=");
401        builder.append(delegate);
402        builder.append(", \nmodifier=");
403        builder.append(modifier);
404        builder.append(", \ncharacterBuffer=");
405        builder.append(characterBuffer);
406        builder.append(", \nnextUnreadCharacterInBuffer=");
407        builder.append(firstModifiableCharacterInBuffer);
408        builder.append(", \nminimumLengthOfLookBehind=");
409        builder.append(minimumLengthOfLookBehind);
410        builder.append(", \nrequestedCapacityOfCharacterBuffer=");
411        builder.append(requestedNumCharactersInBuffer);
412        builder.append(", \nlockedCharacters=");
413        builder.append(numberOfCharactersToSkip);
414        builder.append(", \nendOfStreamHit=");
415        builder.append(endOfStreamHit);
416        builder.append(", \nlastModificationForDebuggingOnly=");
417        builder.append(lastAfterModificationForDebuggingOnly);
418        builder.append("]");
419        return builder.toString();
420    }
