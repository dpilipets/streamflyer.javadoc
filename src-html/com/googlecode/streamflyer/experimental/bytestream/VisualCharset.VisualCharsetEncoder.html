160        public static void main(String[] args) throws Exception {
161            // print the characters
162            // TODO print (1) javadoc, (2) print encoding tables
163    
164            StringBuilder sb = new StringBuilder();
165            for (int i = 0; i < 256; i++) {
166    
167                if (i % 16 == 0 && sb.length() != 0) {
168                    sb.append('\n');
169                }
170    
171                char ch;
172    
173                if (i < 0x20 || i == 0x7F || i == 0x81 || i == 0x8D || i == 0x8F
174                        || i == 0x90 || i == 0x9D || i == 0xA0) {
175                    // 7F DELETE
176                    // 81, 8D, 8F, 90, 9D not defined in windows-1252
177                    // A0 NON BREAKING SPACE (visually not distinguishable from
178                    // SPACE)
179                    // should I replace some dashes as well?
180                    if (i < 0x1F) {
181                        ch = (char) (0x03B0 + i);
182                    }
183                    else {
184                        switch (i) {
185                        case 0x1F:
186                            ch = 0x03A6; // Greek upper-case letter PSI(?)
187                            break;
188                        case 0x7F:
189                            ch = 0x0393; // Greek upper-case letter GAMMA
190                            break;
191                        case 0x81:
192                            ch = 0x0394; // Greek upper-case letter DELTA
193                            break;
194                        case 0x8D:
195                            ch = 0x0398; // Greek upper-case letter OMIKRON
196                            break;
197                        case 0x8F:
198                            ch = 0x039B; // Greek upper-case letter LAMBDA
199                            break;
200                        case 0x90:
201                            ch = 0x039E; // Greek upper-case letter ETA(?)
202                            break;
203                        case 0x9D:
204                            ch = 0x03A0; // Greek upper-case letter PI
205                            break;
206                        case 0xA0:
207                            ch = 0x03A3; // Greek upper-case letter SIGMA
208                            break;
209                        default:
210                            throw new IllegalArgumentException(
211                                    "oops, unknown value: " + i
212                                            + ". typo somewhere?");
213                        }
214                    }
215                }
216                else {
217                    byte b = (byte) i;
218                    // (historical name for windows-1252 is Cp1252)
219                    ch = new String(new byte[] { b }, "windows-1252").charAt(0);
220                }
221    
222                // append character
223                sb.append(ch);
224            }
225    
226            SwingingPrintStream out = new SwingingPrintStream();
227            out.println(sb);
228        }

235                super();

238                textComponent = new JTextArea();
239                // I tried some fonts.

243                textComponent.setFont(new Font("Courier New", Font.PLAIN, 16));
244                frame.getContentPane().add(textComponent);
245                frame.setVisible(true);
246            }

249                textComponent.setText(textComponent.getText() + obj + "\n");
