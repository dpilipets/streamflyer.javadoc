017    package com.googlecode.streamflyer.experimental.stateful;
018    
019    import com.googlecode.streamflyer.core.Modifier;
020    
021    
022    /**
023     * Represents the state of a {@link StatefulModifier}. It carries out the actual
024     * work of the {@link StatefulModifier}. Additionally, it defines the next state
025     * of the {@link StatefulModifier}.
026     * 
027     * @author rwoo
028     * 
029     * @since 14.09.2011
030     */
031    public interface State {
032    
033        /**
034         * Does the same as {@link Modifier#modify(StringBuilder, int, boolean)}
035         * but, additionally, the returned result contains the new state for the
036         * {@link StatefulModifier}.
037         */
038        public StatefulAfterModification modify(StringBuilder characterBuffer,
039                int firstModifiableCharacterInBuffer, boolean endOfStreamHit);
040    
041    
042    }
