017    package com.googlecode.streamflyer.experimental.stateful.util;
018    
019    import java.util.regex.MatchResult;
020    import java.util.regex.Matcher;
021    import java.util.regex.Pattern;
022    
023    import com.googlecode.streamflyer.core.AfterModification;
024    import com.googlecode.streamflyer.experimental.stateful.State;
025    import com.googlecode.streamflyer.experimental.stateful.StatefulAfterModification;
026    import com.googlecode.streamflyer.regex.MatchProcessor;
027    import com.googlecode.streamflyer.regex.MatchProcessorResult;
028    import com.googlecode.streamflyer.regex.OnStreamMatcher;
029    import com.googlecode.streamflyer.regex.OnStreamStandardMatcher;
030    import com.googlecode.streamflyer.regex.RegexModifier;
031    import com.googlecode.streamflyer.thirdparty.ZzzValidate;
032    
033    /**
034     * This state refers to a token (described by a regular expression). If that
035     * token is found, then the state might be changed.
036     * <p>
037     * The default behavior of this class never changes the state nor does it modify
038     * the character buffer. Override
039     * <ul>
040     * <li>{@link #findState(StringBuilder, int, MatchResult)},
041     * <li> {@link #modifyBuffer(StringBuilder, int, MatchResult)},
042     * <li>
043     * {@link #processWithoutMatch(StringBuilder, int, boolean, AfterModification)}
044     * </ul>
045     * in order to modify the behavior.
046     * 
047     * @author rwoo
048     * 
049     * @since 14.09.2011
050     */
051    public class RegexTransitionState implements State, MatchProcessor {
052    
053        private RegexModifier nextTokensRegexModifier;
054    
055        /**
056         * The next state determined by evaluating the result of a match. Null if
057         * there was no match.
058         */
059        private State nextState;
060    
061        /**
062         * @param nextTokensRegexModifier
063         */
064        public RegexTransitionState(String nextTokensRegex) {
065            super();
066    
067            ZzzValidate
068                    .notNull(nextTokensRegex, "nextTokensRegex must not be null");
069    
070            // create onStreamMatcher
071            // TODO inject matcher instead of regex
072            Matcher matcher = Pattern.compile(nextTokensRegex).matcher("");
073            matcher.useTransparentBounds(true);
074            OnStreamMatcher onStreamMatcher = new OnStreamStandardMatcher(matcher);
075    
076            // create nextTokensRegexModifier using 'this' as match processor
077            // TODO remove magic numbers
078            this.nextTokensRegexModifier = new RegexModifier(onStreamMatcher, this,
079                    12, 345);
080        }
081    
082        /**
083         * @see com.googlecode.streamflyer.experimental.stateful.State#modify(java.lang.StringBuilder,
084         *      int, boolean)
085         */
086        @Override
087        public StatefulAfterModification modify(StringBuilder characterBuffer,
088                int firstModifiableCharacterInBuffer, boolean endOfStreamHit) {
089    
090            AfterModification afterModification = nextTokensRegexModifier.modify(
091                    characterBuffer, firstModifiableCharacterInBuffer,
092                    endOfStreamHit);
093    
094            StatefulAfterModification result;
095            if (nextState != null) {
096                // there was a match -> use the next state
097                result = new StatefulAfterModification(afterModification, nextState);
098                nextState = null;
099            }
100            else {
101                // there was no match -> use the existing state
102                afterModification = processWithoutMatch(characterBuffer,
103                        firstModifiableCharacterInBuffer, endOfStreamHit,
104                        afterModification);
105                result = new StatefulAfterModification(afterModification, this);
106            }
107    
108            return result;
109        }
110    
111        /**
112         * @see com.googlecode.streamflyer.regex.MatchProcessor#process(java.lang.StringBuilder,
113         *      int, java.util.regex.MatchResult)
114         */
115        @Override
116        public MatchProcessorResult process(StringBuilder characterBuffer,
117                int firstModifiableCharacterInBuffer, MatchResult matchResult) {
118    
119            // (1) find the next state (this depends on the match result)
120            // TODO use the content of the top-level groups
121            nextState = findState(characterBuffer,
122                    firstModifiableCharacterInBuffer, matchResult);
123    
124            // (2) modify the buffer
125            firstModifiableCharacterInBuffer = modifyBuffer(characterBuffer,
126                    firstModifiableCharacterInBuffer, matchResult);
127    
128            // do we have to change the state?
129            if (!nextState.equals(this)) {
130                // yes, state must be changed -> we must not continue to match with
131                // this object
132                return new MatchProcessorResult(firstModifiableCharacterInBuffer,
133                        false);
134            }
135            else {
136                // no, state must not be changed -> we continue to match with this
137                // object
138                return new MatchProcessorResult(firstModifiableCharacterInBuffer,
139                        true);
140            }
141        }
142    
143        /**
144         * Finds the next state (evaluating the given match result).
145         * <p>
146         * The default implementation returns 'this' state.
147         */
148        protected State findState(StringBuilder characterBuffer,
149                int firstModifiableCharacterInBuffer, MatchResult matchResult) {
150            return this;
151        }
152    
153        /**
154         * Modifies the character buffer if the token was found. Returns the
155         * position of the first character that is modifiable.
156         * <p>
157         * The default implementation does not modify the buffer. Returns the end of
158         * the match as new first modifiable character in the buffer.
159         */
160        protected int modifyBuffer(StringBuilder characterBuffer,
161                int firstModifiableCharacterInBuffer, MatchResult matchResult) {
162            return matchResult.end();
163        }
164    
165        /**
166         * Modifies the character buffer if no token is found.
167         * <p>
168         * The default implementation returns the given {@link AfterModification}.
169         */
170        protected AfterModification processWithoutMatch(
171                StringBuilder characterBuffer,
172                int firstModifiableCharacterInBuffer, boolean endOfStreamHit,
173                AfterModification afterModification) {
174            return afterModification;
175        }
176    
177    
178    }
