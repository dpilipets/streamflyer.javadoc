017    package com.googlecode.streamflyer.regex;
018    
019    import java.util.regex.MatchResult;
020    
021    import com.googlecode.streamflyer.core.Modifier;
022    
023    /**
024     * This interface is used by {@link RegexModifier}. This interface defines what
025     * to do if the regular expression matches some text in the stream.
026     * 
027     * @author rwoo
028     * @since 18.06.2011
029     */
030    public interface MatchProcessor {
031    
032        /**
033         * Processes the given match. The match processor is allowed to modify the
034         * given character buffer but it must not modify the characters before the
035         * first modifiable character in the buffer because the characters before
036         * that position are considered unmodifiable.
037         * 
038         * @param characterBuffer Must not be <code>null</code>.
039         * @param firstModifiableCharacterInBuffer See the parameter with the same
040         *        name in {@link Modifier#modify(StringBuilder, int, boolean)}.
041         * @param matchResult The match that is found in the given character buffer.
042         *        Must not be <code>null</code>.
043         * @return Returns an {@link MatchProcessorResult object} that describes
044         *         where to continue the matching.
045         */
046        public MatchProcessorResult process(StringBuilder characterBuffer,
047                int firstModifiableCharacterInBuffer, MatchResult matchResult);
048    
049    }
