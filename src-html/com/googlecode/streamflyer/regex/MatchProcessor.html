017    package com.googlecode.streamflyer.regex;
018    
019    import java.util.regex.MatchResult;
020    
021    /**
022     * Processes a match provided by {@link RegexModifier}.
023     * 
024     * @author rwoo
025     * 
026     * @since 18.06.2011
027     */
028    public interface MatchProcessor {
029    
030        /**
031         * Processes the given match. The match processor is allowed to modify the
032         * given character buffer but it must not modify the characters before the
033         * first modifiable character in the buffer because the characters before
034         * that position are considered unmodifiable.
035         * 
036         * @param characterBuffer Must not be <code>null</code>.
037         * @param firstModifiableCharacterInBuffer TODO
038         * @param matchResult The match that is found in the given character buffer.
039         *        Must not be <code>null</code>.
040         * @return Returns
041         *         <ol>
042         *         <li>a position in the character buffer. This position must not be
043         *         smaller than the first modifiable character in the buffer. The
044         *         characters before the returned position shall be considered
045         *         unmodifiable by the {@link RegexModifier}.
046         *         <li>true if the matching shall be continued using the characters
047         *         already available in the buffer.
048         *         </ol>
049         */
050        public MatchProcessorResult process(StringBuilder characterBuffer,
051                int firstModifiableCharacterInBuffer, MatchResult matchResult);
052    
053    }
