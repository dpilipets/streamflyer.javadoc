017    package com.googlecode.streamflyer.regex;
018    
019    import java.util.regex.MatchResult;
020    import java.util.regex.Matcher;
021    import java.util.regex.Pattern;
022    
023    import com.googlecode.streamflyer.core.AfterModification;
024    import com.googlecode.streamflyer.core.Modifier;
025    import com.googlecode.streamflyer.util.ModificationFactory;
026    import com.googlecode.streamflyer.util.ModifyingReaderFactory;
027    
028    /**
029     * Finds text that matches a given regular expression. The match is processed by
030     * the given {@link MatchProcessor}.
031     * <p>
032     * TODO Memory consumption in detail including greedy operators
033     * <p>
034     * TODO explain lethal regex like "(look-behind expression)()".
035     * 
036     * @author rwoo
037     * @since 18.06.2011
038     */
039    public class RegexModifier implements Modifier {
040    
041        //
042        // injected
043        //
044    
045        protected ModificationFactory factory;
046    
047        protected MatchProcessor matchProcessor;
048    
049        /**
050         * The compiled representation of a regular expression. If the regular
051         * expression matches, then a modification shall be carried out.
052         */
053        protected OnStreamMatcher matcher;
054    
055        protected int newNumberOfChars = -1;
056    
057    
058        //
059        // state
060        //
061    
062        //
063        //
064        //
065        /**
066         * @deprecated Use
067         *             {@link #RegexModifier(OnStreamMatcher, MatchProcessor, int, int)}
068         *             or {@link ModifyingReaderFactory} instead.
069         */
070        @Deprecated
071        public RegexModifier(String regex, int flags, String replacement,
072                int minimumLengthOfLookBehind, int newNumberOfChars) {
073    
074            Matcher jdkMatcher = Pattern.compile(regex, flags).matcher("");
075            jdkMatcher.useTransparentBounds(true);
076            init(new OnStreamStandardMatcher(jdkMatcher), new ReplacingProcessor(
077                    replacement), minimumLengthOfLookBehind, newNumberOfChars);
078        }
079    
080        /**
081         * @param matcher
082         * @param replacement
083         * @param minimumLengthOfLookBehind
084         * @param newNumberOfChars This should not be smaller than the length of the
085         *        characters sequence the {@link #pattern} needs to match properly.
086         *        In case you want to match more than once, the value should be
087         *        higher.
088         */
089        public RegexModifier(OnStreamMatcher matcher,
090                MatchProcessor matchProcessor, int minimumLengthOfLookBehind,
091                int newNumberOfChars) {
092    
093            init(matcher, matchProcessor, minimumLengthOfLookBehind,
094                    newNumberOfChars);
095        }
096    
097        @SuppressWarnings("hiding")
098        protected void init(OnStreamMatcher matcher, MatchProcessor matchProcessor,
099                int minimumLengthOfLookBehind, int newNumberOfChars) {
100    
101            this.factory = new ModificationFactory(minimumLengthOfLookBehind,
102                    newNumberOfChars);
103            this.matchProcessor = matchProcessor;
104            this.matcher = matcher;
105            this.newNumberOfChars = newNumberOfChars;
106        }
107    
108    
109        //
110        // interface Modifier
111        //
112    
113        /**
114         * @see com.googlecode.streamflyer.core.Modifier#modify(java.lang.StringBuilder,
115         *      int, boolean)
116         */
117        @Override
118        public AfterModification modify(StringBuilder characterBuffer,
119                int firstModifiableCharacterInBuffer, boolean endOfStreamHit) {
120    
121            // the first position we will match from.
122            Integer minFrom = null;
123    
124            while (true) {
125    
126                // determine the range [minFrom, maxFrom] that will contain the
127                // first character of the matching string
128    
129                if (minFrom == null) {
130                    minFrom = firstModifiableCharacterInBuffer;
131                }
132    
133                int maxFrom = firstModifiableCharacterInBuffer + newNumberOfChars;
134                if (maxFrom > characterBuffer.length()) {
135                    // we have to restrict maxFrom in order to prevent that the
136                    // requested number of characters increases more and more
137    
138                    // this is NOT set to characterBuffer.length() -1 by intention
139                    // as a regular expression might match on the zero length string
140                    // (but with positive look-behind)!
141                    maxFrom = characterBuffer.length();
142                }
143    
144                // find first match
145                // (as the match processor might have modified the buffer, we reset
146                // the matcher inside the loop instead of outside of the loop)
147                matcher.reset(characterBuffer);
148                boolean matchFound = matcher.findUnlessHitEnd(minFrom, maxFrom);
149    
150                if (matchFound) {
151                    // we found a match
152    
153                    // could change this positive match into a negative one
154                    // (matcher.requireEnd()) or into a longer one (greedy
155                    // operator)?
156                    if (matcher.hitEnd() && !endOfStreamHit) {
157                        // (UC11000) yes, it could -> we need more input
158    
159                        int numberOfCharactersToSkip = matcher.lastFrom()
160                                - firstModifiableCharacterInBuffer;
161    
162                        // read more input (skip some characters if possible)
163                        return factory.fetchMoreInput(numberOfCharactersToSkip,
164                                characterBuffer, firstModifiableCharacterInBuffer,
165                                endOfStreamHit);
166                    }
167                    else {
168                        // no -> thus we can use this match
169    
170                        // process the match
171                        MatchResult matchResult = matcher; // TODO .toMatchResult()?
172                        // TODO please review: should I pass
173                        // firstModifiableCharacterInBuffer instead of minFrom?
174                        MatchProcessorResult matchProcessorResult = matchProcessor
175                                .process(characterBuffer, minFrom, matchResult);
176                        minFrom = matchProcessorResult
177                                .getFirstModifiableCharacterInBuffer();
178    
179                        // match again? (even for minFrom == maxFrom we try a match)
180                        if (minFrom <= maxFrom
181                                && matchProcessorResult.isContinueMatching()) {
182                            // (UC12100)
183    
184                            // We try the next match on the modified input, i.e.
185                            // not match only once -> next loop
186                            continue;
187                        }
188                        else {
189    
190                            // (UC12200)
191    
192                            // TODO please doc, this is IMPORTANT!
193    
194                            int numberOfCharactersToSkip = minFrom
195                                    - firstModifiableCharacterInBuffer;
196    
197                            if (numberOfCharactersToSkip == 0) {
198                                // TODO passing false for endOfStreamHit is ugly!!!
199                                // we should offer a new method in
200                                // ModificationFactory, something like
201                                // continueAfterModification(...) that chooses the
202                                // appropriate action. the following code is always
203                                // a MODIFY_AGAIN_IMMEDIATELY
204                                return factory.fetchMoreInput(
205                                        numberOfCharactersToSkip, characterBuffer,
206                                        firstModifiableCharacterInBuffer, false);
207                            }
208                            else {
209                                return factory.skipOrStop(numberOfCharactersToSkip,
210                                        characterBuffer,
211                                        firstModifiableCharacterInBuffer,
212                                        endOfStreamHit);
213                            }
214                        }
215    
216                    }
217                }
218                else {
219                    // we haven't found a match
220    
221                    // did we match at least one character in range [from, maxFrom]?
222                    // TODO matcher.lastFrom() < maxFrom or matcher.lastFrom() <=
223                    // maxFrom? add a test!
224                    if (matcher.lastFrom() <= maxFrom) {
225                        // yes, we are matching something (at least one character)
226    
227                        // we did no match and we hit the end of the input from a
228                        // position that is not the at the end of the input.
229                        // Therefore, we must assume that more input could find a
230                        // match
231    
232                        // can we get more input?
233                        if (endOfStreamHit) {
234                            // (UC21100) no, in the entire stream we will not found
235                            // more matches that start in range [from,
236                            // maxFrom] -> skip the characters from range [from,
237                            // maxFrom]
238    
239                            int numberOfCharactersToSkip = maxFrom
240                                    - firstModifiableCharacterInBuffer;
241                            return factory.skipOrStop(numberOfCharactersToSkip,
242                                    characterBuffer,
243                                    firstModifiableCharacterInBuffer,
244                                    endOfStreamHit);
245    
246                        }
247                        else {
248                            // yes > we should fetch more input (because end
249                            // of stream is not hit yet)
250    
251                            int numberOfCharactersToSkip = matcher.lastFrom()
252                                    - firstModifiableCharacterInBuffer;
253                            return factory.fetchMoreInput(numberOfCharactersToSkip,
254                                    characterBuffer,
255                                    firstModifiableCharacterInBuffer,
256                                    endOfStreamHit);
257    
258                        }
259    
260                    }
261                    else { // matcher.lastFrom() == maxFrom + 1
262    
263                        // (UC22000) no, we are matching not a single character
264    
265                        // -> skip the characters from range [from, maxFrom]
266                        int numberOfCharactersToSkip = maxFrom
267                                - firstModifiableCharacterInBuffer;
268                        return factory.skipOrStop(numberOfCharactersToSkip,
269                                characterBuffer, firstModifiableCharacterInBuffer,
270                                endOfStreamHit);
271                    }
272                }
273            }
274    
275        }
276    
277        /**
278         * @see java.lang.Object#toString()
279         */
280        @Override
281        public String toString() {
282            StringBuilder builder = new StringBuilder();
283            builder.append("RegexModifier [\nfactory=");
284            builder.append(factory);
285            builder.append(", \nreplacement=");
286            builder.append(matchProcessor);
287            builder.append(", \nmatcher=");
288            builder.append(matcher);
289            builder.append(", \nnewNumberOfChars=");
290            builder.append(newNumberOfChars);
291            builder.append("]");
292            return builder.toString();
293        }
