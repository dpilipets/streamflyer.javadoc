017    package com.googlecode.streamflyer.regex;
018    
019    import java.util.ArrayList;
020    import java.util.List;
021    import java.util.regex.MatchResult;
022    import java.util.regex.Matcher;
023    
024    import com.googlecode.streamflyer.thirdparty.ZzzValidate;
025    
026    /**
027     * This {@link MatchProcessor} replaces the matching string (given by the
028     * {@link MatchResult}) with the configured replacement.
029     * <p>
030     * The replacement can be defined with the same syntax as defined by
031     * {@link Matcher#appendReplacement(StringBuffer, String)}
032     * <p>
033     * EXAMPLE: <code><pre>
034    new ReplacingProcessor("aaa") // replaces each match with "aaa"
035    new ReplacingProcessor("$2") // replaces each match with the content of the second {@link MatchResult#group() group} of the match
036    new ReplacingProcessor("aaa$2bbb$1ccc") // replaces each match with "aaa$2bbb$1ccc"
037                               // wherein $2 and $1 are replaced with the content
038                               // of the second and the first group of the match
039    </pre></code>
040     * 
041     * @author rwoo
042     * @since 18.06.2011
043     */
044    public class ReplacingProcessor implements MatchProcessor {
045    
046        //
047        // injected properties
048        //
049    
050        /**
051         * This list contains Strings and Numbers. The latter ones refer to groups.
052         */
053        private List<Object> parts = null;
054    
055        /**
056         * If not null, then the replacement do not uses groups. Then this property
057         * contains the entire replacement.
058         */
059        private CharSequence replacementWithoutGroupReferences;
060    
061        //
062        // constructors
063        //
064    
065        /**
066         * For unit tests only.
067         */
068        public ReplacingProcessor() {
069            // nothing to do here
070        }
071    
072        /**
073         * @param replacement
074         */
075        public ReplacingProcessor(String replacement) {
076            super();
077    
078            // validate arguments
079            ZzzValidate.notNull(replacement, "replacement must not be null");
080    
081            // parse replacement
082            this.parts = parseReplacement(replacement);
083    
084            if (parts.size() == 1 && (parts.get(0) instanceof CharSequence)) {
085    
086                this.replacementWithoutGroupReferences = (CharSequence) parts
087                        .get(0);
088    
089            }
090            else {
091    
092                this.replacementWithoutGroupReferences = null;
093    
094            }
095        }
096    
097    
098        /**
099         * Package-private for JUnit-tests.
100         * 
101         * @return Returns the parts of the matchProcessor
102         */
103        List<Object> parseReplacement(String replacement) {
104    
105            List<Object> compiledReplacement = new ArrayList<Object>();
106    
107            // we look for escaped literals and references to groups
108            int position = 0;
109            StringBuilder notGroupReference = new StringBuilder();
110    
111            while (position < replacement.length()) {
112                char ch = replacement.charAt(position);
113    
114                // reference to group?
115                if (ch == '$') {
116    
117                    if (notGroupReference.length() != 0) {
118                        compiledReplacement.add(notGroupReference.toString());
119                        notGroupReference.setLength(0);
120                    }
121    
122                    position++;
123    
124                    int groupNumberStartPosition = position;
125                    if (groupNumberStartPosition == replacement.length()) {
126                        throw new IllegalArgumentException("group reference $ "
127                                + "without number at the end " + "of the "
128                                + "replacement string (" + replacement + ")");
129                    }
130    
131                    ch = replacement.charAt(position);
132    
133                    while ('0' <= ch && ch <= '9') {
134                        position++;
135                        if (position == replacement.length()) {
136                            break;
137                        }
138                        ch = replacement.charAt(position);
139                    }
140                    // position is the position of the first character that does not
141                    // belong to the group reference
142    
143                    if (position == groupNumberStartPosition) {
144                        throw new IllegalArgumentException(
145                                "invalid reference to group at position "
146                                        + position + " in replacement string "
147                                        + replacement.substring(0, position) + "[]"
148                                        + replacement.substring(position));
149                    }
150    
151                    String groupNumberString = replacement.substring(
152                            groupNumberStartPosition, position);
153                    int groupNumber = Integer.parseInt(groupNumberString);
154    
155                    compiledReplacement.add(groupNumber);
156                }
157                // escaped literal?
158                else if (ch == '\\') {
159    
160                    // skip the backslash
161                    position++;
162                    ch = replacement.charAt(position);
163    
164                    notGroupReference.append(ch);
165                    position++;
166                }
167                // this is a normal character
168                else {
169    
170                    notGroupReference.append(ch);
171                    position++;
172                }
173            }
174    
175            if (notGroupReference.length() != 0) {
176                compiledReplacement.add(notGroupReference.toString());
177            }
178    
179            return compiledReplacement;
180        }
181    
182        /**
183         * Replaces the match as given by the {@link MatchResult} with the
184         * configured replacement.
185         * 
186         * @see com.googlecode.streamflyer.regex.MatchProcessor#process(java.lang.StringBuilder,
187         *      int, java.util.regex.MatchResult)
188         */
189        @Override
190        public MatchProcessorResult process(StringBuilder characterBuffer,
191                int firstModifiableCharacterInBuffer, MatchResult matchResult) {
192    
193            int start = matchResult.start();
194            int end = matchResult.end();
195    
196            // if the empty string is matched, then we increase the position
197            // to continue from to avoid endless loops
198            // (compare to Matcher.find() where we see the following code:
199            // int i = last; if(i == first) i++;
200            // in words: set the *from* for the next match at the
201            // end of the last match. if this is equal to the start
202            // of the last match (a match on the empty string(, then
203            // increase the *from* to avoid endless loops)
204            int offset = start == end ? 1 : 0;
205    
206            if (replacementWithoutGroupReferences != null) {
207    
208                characterBuffer.delete(start, end);
209                characterBuffer.insert(start, replacementWithoutGroupReferences);
210                return new MatchProcessorResult(start
211                        + replacementWithoutGroupReferences.length() + offset, true);
212    
213            }
214            else {
215    
216                // (1) create replacement
217                // (one could set a reasonable value for the initial size of the
218                // string builder somewhere)
219                StringBuilder replacement = new StringBuilder();
220    
221                for (Object part : parts) {
222                    if (part instanceof Integer) {
223                        // append the value of the referred group
224                        replacement.append(matchResult.group((Integer) part));
225                    }
226                    else {
227                        // append the static part
228                        replacement.append(part); // part is instance of String
229                    }
230                }
231    
232                // (2) replace the match with the replacement
233                characterBuffer.delete(start, end);
234                characterBuffer.insert(start, replacement);
235    
236                return new MatchProcessorResult(start + replacement.length()
237                        + offset, true);
238            }
239    
240        }
241    
242    }
