016package com.googlecode.streamflyer.regex.addons.stateful;
017
018import java.util.regex.MatchResult;
019
020import com.googlecode.streamflyer.regex.MatchProcessorResult;
021
022/**
023 * A transition guard can be used to stop a {@link Transitions}. By default, transitions are not stopped. Use a subclass
024 * when some transitions shall be stopped.
025 * <p>
026 * Apart from that, the guard is a good place to execute logic that shall be executed during the transition, before the
027 * new state is reached. The logic must not be specific to this transition.
028 * 
029 * @author rwoo
030 * @since 1.1.0
031 */
032public class TransitionGuard {
033
034    /**
035     * @param nextState
036     * @param characterBuffer
037     * @param firstModifiableCharacterInBuffer
038     * @param matchResult
039     * @return Returns null if the transition shall be executed. Returns a {@link MatchProcessorResult} that shall be
040     *         returned if the transition shall be stopped. If not overwritten, this method returns null.
041     */
042    public MatchProcessorResult stopTransition(State nextState, StringBuilder characterBuffer,
043            int firstModifiableCharacterInBuffer, MatchResult matchResult) {
044        // by default we don't stop the transition
045        return null;
046    }
047
048}
