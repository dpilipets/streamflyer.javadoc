028    /**
029     * An engine that performs match operations on a {@link java.lang.CharSequence
030     * </code>character sequence<code>} by interpreting a {@link Pattern}.
031     * <p>
032     * A matcher is created from a pattern by invoking the pattern's
033     * {@link Pattern#matcher matcher} method. Once created, a matcher can be used
034     * to perform three different kinds of match operations:
035     * <ul>
036     * <li>
037     * <p>
038     * The {@link #matches matches} method attempts to match the entire input
039     * sequence against the pattern.
040     * </p>
041     * </li>
042     * <li>
043     * <p>
044     * The {@link #lookingAt lookingAt} method attempts to match the input sequence,
045     * starting at the beginning, against the pattern.
046     * </p>
047     * </li>
048     * <li>
049     * <p>
050     * The {@link #find find} method scans the input sequence looking for the next
051     * subsequence that matches the pattern.
052     * </p>
053     * </li>
054     * </ul>
055     * <p>
056     * Each of these methods returns a boolean indicating success or failure. More
057     * information about a successful match can be obtained by querying the state of
058     * the matcher.
059     * <p>
060     * A matcher finds matches in a subset of its input called the <i>region</i>. By
061     * default, the region contains all of the matcher's input. The region can be
062     * modified via the{@link #region region} method and queried via the
063     * {@link #regionStart regionStart} and {@link #regionEnd regionEnd} methods.
064     * The way that the region boundaries interact with some pattern constructs can
065     * be changed. See {@link #useAnchoringBounds useAnchoringBounds} and
066     * {@link #useTransparentBounds useTransparentBounds} for more details.
067     * <p>
068     * This class also defines methods for replacing matched subsequences with new
069     * strings whose contents can, if desired, be computed from the match result.
070     * The {@link #appendReplacement appendReplacement} and {@link #appendTail
071     * appendTail} methods can be used in tandem in order to collect the result into
072     * an existing string buffer, or the more convenient {@link #replaceAll
073     * replaceAll} method can be used to create a string in which every matching
074     * subsequence in the input sequence is replaced.
075     * <p>
076     * The explicit state of a matcher includes the start and end indices of the
077     * most recent successful match. It also includes the start and end indices of
078     * the input subsequence captured by each <a href="Pattern.html#cg">capturing
079     * group</a> in the pattern as well as a total count of such subsequences. As a
080     * convenience, methods are also provided for returning these captured
081     * subsequences in string form.
082     * <p>
083     * The explicit state of a matcher is initially undefined; attempting to query
084     * any part of it before a successful match will cause an
085     * {@link IllegalStateException} to be thrown. The explicit state of a matcher
086     * is recomputed by every match operation.
087     * <p>
088     * The implicit state of a matcher includes the input character sequence as well
089     * as the <i>append position</i>, which is initially zero and is updated by the
090     * {@link #appendReplacement appendReplacement} method.
091     * <p>
092     * A matcher may be reset explicitly by invoking its {@link #reset()} method or,
093     * if a new input sequence is desired, its
094     * {@link #reset(java.lang.CharSequence) reset(CharSequence)} method. Resetting
095     * a matcher discards its explicit state information and sets the append
096     * position to zero.
097     * <p>
098     * Instances of this class are not safe for use by multiple concurrent threads.
099     * </p>
100     * 
101     * @author Mike McCloskey
102     * @author Mark Reinhold
103     * @author JSR-51 Expert Group
104     * @version 1.65, 10/03/23
105     * @since 1.4
106     * @spec JSR-51
107     */
108    
109    @SuppressWarnings({ "hiding", "unused" })
110    // NEW @SuppressWarnings ADDED BY rwoo
111    public final class Matcher implements MatchResult {
112    
113        /**
114         * The Pattern object that created this Matcher.
115         */
116        Pattern parentPattern;
117    
118        /**
119         * The storage used by groups. They may contain invalid values if a group
120         * was skipped during the matching.
121         */
122        int[] groups;
123    
124        /**
125         * The range within the sequence that is to be matched. Anchors will match
126         * at these "hard" boundaries. Changing the region changes these values.
127         */
128        int from, to;
129    
130        /**
131         * Lookbehind uses this value to ensure that the subexpression match ends at
132         * the point where the lookbehind was encountered.
133         */
134        int lookbehindTo;
135    
136        /**
137         * The original string being matched.
138         */
139        CharSequence text;
140    
141        /**
142         * Matcher state used by the last node. NOANCHOR is used when a match does
143         * not have to consume all of the input. ENDANCHOR is the mode used for
144         * matching all the input.
145         */
146        static final int ENDANCHOR = 1;
147    
148        static final int NOANCHOR = 0;
149    
150        int acceptMode = NOANCHOR;
151    
152        /**
153         * The range of string that last matched the pattern. If the last match
154         * failed then first is -1; last initially holds 0 then it holds the index
155         * of the end of the last match (which is where the next search starts).
156         */
157        int first = -1, last = 0;
158    
159        /**
160         * The end index of what matched in the last match operation.
161         */
162        int oldLast = -1;
163    
164        /**
165         * The index of the last position appended in a substitution.
166         */
167        int lastAppendPosition = 0;
168    
169        /**
170         * Storage used by nodes to tell what repetition they are on in a pattern,
171         * and where groups begin. The nodes themselves are stateless, so they rely
172         * on this field to hold state during a match.
173         */
174        int[] locals;
175    
176        /**
177         * Boolean indicating whether or not more input could change the results of
178         * the last match. If hitEnd is true, and a match was found, then more input
179         * might cause a different match to be found. If hitEnd is true and a match
180         * was not found, then more input could cause a match to be found. If hitEnd
181         * is false and a match was found, then more input will not change the
182         * match. If hitEnd is false and a match was not found, then more input will
183         * not cause a match to be found.
184         */
185        boolean hitEnd;
186    
187        /**
188         * Boolean indicating whether or not more input could change a positive
189         * match into a negative one. If requireEnd is true, and a match was found,
190         * then more input could cause the match to be lost. If requireEnd is false
191         * and a match was found, then more input might change the match but the
192         * match won't be lost. If a match was not found, then requireEnd has no
193         * meaning.
194         */
195        boolean requireEnd;
196    
197        /**
198         * If transparentBounds is true then the boundaries of this matcher's region
199         * are transparent to lookahead, lookbehind, and boundary matching
200         * constructs that try to see beyond them.
201         */
202        boolean transparentBounds = false;
203    
204        /**
205         * If anchoringBounds is true then the boundaries of this matcher's region
206         * match anchors such as ^ and $.
207         */
208        boolean anchoringBounds = true;
209    
210        /**
211         * No default constructor.
212         */
213        Matcher() {
214            //
215        }
216    
217        /**
218         * All matchers have the state used by Pattern during a match.
219         */
220        Matcher(Pattern parent, CharSequence text) {
221            this.parentPattern = parent;
222            this.text = text;
223    
224            // Allocate state storage
225            int parentGroupCount = Math.max(parent.capturingGroupCount, 10);
226            groups = new int[parentGroupCount * 2];
227            locals = new int[parent.localCount];
228    
229            // Put fields into initial states
230            reset();
231        }
232    
233        /**
234         * Returns the pattern that is interpreted by this matcher.
235         * 
236         * @return The pattern for which this matcher was created
237         */
238        public Pattern pattern() {
239            return parentPattern;
240        }
241    
242        /**
243         * Returns the match state of this matcher as a {@link MatchResult}. The
244         * result is unaffected by subsequent operations performed upon this
245         * matcher.
246         * 
247         * @return a <code>MatchResult</code> with the state of this matcher
248         * @since 1.5
249         */
250        public MatchResult toMatchResult() {
251            Matcher result = new Matcher(this.parentPattern, text.toString());
252            result.first = this.first;
253            result.last = this.last;
254            result.groups = (this.groups.clone());
255            return result;
256        }
257    
258        /**
259         * Changes the <tt>Pattern</tt> that this <tt>Matcher</tt> uses to find
260         * matches with.
261         * <p>
262         * This method causes this matcher to lose information about the groups of
263         * the last match that occurred. The matcher's position in the input is
264         * maintained and its last append position is unaffected.
265         * </p>
266         * 
267         * @param newPattern The new pattern used by this matcher
268         * @return This matcher
269         * @throws IllegalArgumentException If newPattern is <tt>null</tt>
270         * @since 1.5
271         */
272        public Matcher usePattern(Pattern newPattern) {
273            if (newPattern == null) {
274                throw new IllegalArgumentException("Pattern cannot be null");
275            }
276            parentPattern = newPattern;
277    
278            // Reallocate state storage
279            int parentGroupCount = Math.max(newPattern.capturingGroupCount, 10);
280            groups = new int[parentGroupCount * 2];
281            locals = new int[newPattern.localCount];
282            for (int i = 0; i < groups.length; i++) {
283                groups[i] = -1;
284            }
285            for (int i = 0; i < locals.length; i++) {
286                locals[i] = -1;
287            }
288            return this;
289        }
290    
291        /**
292         * Resets this matcher.
293         * <p>
294         * Resetting a matcher discards all of its explicit state information and
295         * sets its append position to zero. The matcher's region is set to the
296         * default region, which is its entire character sequence. The anchoring and
297         * transparency of this matcher's region boundaries are unaffected.
298         * 
299         * @return This matcher
300         */
301        public Matcher reset() {
302            first = -1;
303            last = 0;
304            oldLast = -1;
305            for (int i = 0; i < groups.length; i++) {
306                groups[i] = -1;
307            }
308            for (int i = 0; i < locals.length; i++) {
309                locals[i] = -1;
310            }
311            lastAppendPosition = 0;
312            from = 0;
313            to = getTextLength();
314            return this;
315        }
316    
317        /**
318         * Resets this matcher with a new input sequence.
319         * <p>
320         * Resetting a matcher discards all of its explicit state information and
321         * sets its append position to zero. The matcher's region is set to the
322         * default region, which is its entire character sequence. The anchoring and
323         * transparency of this matcher's region boundaries are unaffected.
324         * 
325         * @param input The new input character sequence
326         * @return This matcher
327         */
328        public Matcher reset(CharSequence input) {
329            text = input;
330            return reset();
331        }

333        /**
334         * Returns the start index of the previous match. </p>
335         * 
336         * @return The index of the first character matched
337         * @throws IllegalStateException If no match has yet been attempted, or if
338         *         the previous match operation failed
339         */
340        @Override
341        public int start() {
342            if (first < 0) {
343                throw new IllegalStateException("No match available");
344            }
345            return first;
346        }
347    
348        /**
349         * Returns the start index of the subsequence captured by the given group
350         * during the previous match operation.
351         * <p>
352         * <a href="Pattern.html#cg">Capturing groups</a> are indexed from left to
353         * right, starting at one. Group zero denotes the entire pattern, so the
354         * expression <i>m.</i><tt>start(0)</tt> is equivalent to <i>m.</i>
355         * <tt>start()</tt>.
356         * </p>
357         * 
358         * @param group The index of a capturing group in this matcher's pattern
359         * @return The index of the first character captured by the group, or
360         *         <tt>-1</tt> if the match was successful but the group itself did
361         *         not match anything
362         * @throws IllegalStateException If no match has yet been attempted, or if
363         *         the previous match operation failed
364         * @throws IndexOutOfBoundsException If there is no capturing group in the
365         *         pattern with the given index
366         */
367        @Override
368        public int start(int group) {
369            if (first < 0) {
370                throw new IllegalStateException("No match available");
371            }
372            if (group > groupCount()) {
373                throw new IndexOutOfBoundsException("No group " + group);
374            }
375            return groups[group * 2];
376        }
377    
378        /**
379         * Returns the offset after the last character matched. </p>
380         * 
381         * @return The offset after the last character matched
382         * @throws IllegalStateException If no match has yet been attempted, or if
383         *         the previous match operation failed
384         */
385        @Override
386        public int end() {
387            if (first < 0) {
388                throw new IllegalStateException("No match available");
389            }
390            return last;
391        }
392    
393        /**
394         * Returns the offset after the last character of the subsequence captured
395         * by the given group during the previous match operation.
396         * <p>
397         * <a href="Pattern.html#cg">Capturing groups</a> are indexed from left to
398         * right, starting at one. Group zero denotes the entire pattern, so the
399         * expression <i>m.</i><tt>end(0)</tt> is equivalent to <i>m.</i>
400         * <tt>end()</tt>.
401         * </p>
402         * 
403         * @param group The index of a capturing group in this matcher's pattern
404         * @return The offset after the last character captured by the group, or
405         *         <tt>-1</tt> if the match was successful but the group itself did
406         *         not match anything
407         * @throws IllegalStateException If no match has yet been attempted, or if
408         *         the previous match operation failed
409         * @throws IndexOutOfBoundsException If there is no capturing group in the
410         *         pattern with the given index
411         */
412        @Override
413        public int end(int group) {
414            if (first < 0) {
415                throw new IllegalStateException("No match available");
416            }
417            if (group > groupCount()) {
418                throw new IndexOutOfBoundsException("No group " + group);
419            }
420            return groups[group * 2 + 1];
421        }
422    
423        /**
424         * Returns the input subsequence matched by the previous match.
425         * <p>
426         * For a matcher <i>m</i> with input sequence <i>s</i>, the expressions
427         * <i>m.</i><tt>group()</tt> and <i>s.</i><tt>substring(</tt><i>m.</i>
428         * <tt>start(),</tt>&nbsp;<i>m.</i><tt>end())</tt> are equivalent.
429         * </p>
430         * <p>
431         * Note that some patterns, for example <tt>a*</tt>, match the empty string.
432         * This method will return the empty string when the pattern successfully
433         * matches the empty string in the input.
434         * </p>
435         * 
436         * @return The (possibly empty) subsequence matched by the previous match,
437         *         in string form
438         * @throws IllegalStateException If no match has yet been attempted, or if
439         *         the previous match operation failed
440         */
441        @Override
442        public String group() {
443            return group(0);
444        }
445    
446        /**
447         * Returns the input subsequence captured by the given group during the
448         * previous match operation.
449         * <p>
450         * For a matcher <i>m</i>, input sequence <i>s</i>, and group index
451         * <i>g</i>, the expressions <i>m.</i><tt>group(</tt><i>g</i><tt>)</tt> and
452         * <i>s.</i><tt>substring(</tt><i>m.</i><tt>start(</tt><i>g</i><tt>),</tt>
453         * &nbsp;<i>m.</i><tt>end(</tt><i>g</i><tt>))</tt> are equivalent.
454         * </p>
455         * <p>
456         * <a href="Pattern.html#cg">Capturing groups</a> are indexed from left to
457         * right, starting at one. Group zero denotes the entire pattern, so the
458         * expression <tt>m.group(0)</tt> is equivalent to <tt>m.group()</tt>.
459         * </p>
460         * <p>
461         * If the match was successful but the group specified failed to match any
462         * part of the input sequence, then <tt>null</tt> is returned. Note that
463         * some groups, for example <tt>(a*)</tt>, match the empty string. This
464         * method will return the empty string when such a group successfully
465         * matches the empty string in the input.
466         * </p>
467         * 
468         * @param group The index of a capturing group in this matcher's pattern
469         * @return The (possibly empty) subsequence captured by the group during the
470         *         previous match, or <tt>null</tt> if the group failed to match
471         *         part of the input
472         * @throws IllegalStateException If no match has yet been attempted, or if
473         *         the previous match operation failed
474         * @throws IndexOutOfBoundsException If there is no capturing group in the
475         *         pattern with the given index
476         */
477        @Override
478        public String group(int group) {
479            if (first < 0) {
480                throw new IllegalStateException("No match found");
481            }
482            if (group < 0 || group > groupCount()) {
483                throw new IndexOutOfBoundsException("No group " + group);
484            }
485            if ((groups[group * 2] == -1) || (groups[group * 2 + 1] == -1)) {
486                return null;
487            }
488            return getSubSequence(groups[group * 2], groups[group * 2 + 1])
489                    .toString();
490        }
491    
492        /**
493         * Returns the number of capturing groups in this matcher's pattern.
494         * <p>
495         * Group zero denotes the entire pattern by convention. It is not included
496         * in this count.
497         * <p>
498         * Any non-negative integer smaller than or equal to the value returned by
499         * this method is guaranteed to be a valid group index for this matcher.
500         * </p>
501         * 
502         * @return The number of capturing groups in this matcher's pattern
503         */
504        @Override
505        public int groupCount() {
506            return parentPattern.capturingGroupCount - 1;
507        }

509        /**
510         * Attempts to match the entire region against the pattern.
511         * <p>
512         * If the match succeeds then more information can be obtained via the
513         * <tt>start</tt>, <tt>end</tt>, and <tt>group</tt> methods.
514         * </p>
515         * 
516         * @return <tt>true</tt> if, and only if, the entire region sequence matches
517         *         this matcher's pattern
518         */
519        public boolean matches() {
520            return match(from, ENDANCHOR);
521        }
522    
523        /**
524         * Attempts to find the next subsequence of the input sequence that matches
525         * the pattern.
526         * <p>
527         * This method starts at the beginning of this matcher's region, or, if a
528         * previous invocation of the method was successful and the matcher has not
529         * since been reset, at the first character not matched by the previous
530         * match.
531         * <p>
532         * If the match succeeds then more information can be obtained via the
533         * <tt>start</tt>, <tt>end</tt>, and <tt>group</tt> methods.
534         * </p>
535         * 
536         * @return <tt>true</tt> if, and only if, a subsequence of the input
537         *         sequence matches this matcher's pattern
538         */
539        public boolean find() {
540            int nextSearchIndex = last;
541            if (nextSearchIndex == first) {
542                nextSearchIndex++;
543            }
544    
545            // If next search starts before region, start it at region
546            if (nextSearchIndex < from) {
547                nextSearchIndex = from;
548            }
549    
550            // If next search starts beyond region then it fails
551            if (nextSearchIndex > to) {
552                for (int i = 0; i < groups.length; i++) {
553                    groups[i] = -1;
554                }
555                return false;
556            }
557            return search(nextSearchIndex);
558        }
559    
560        /**
561         * Resets this matcher and then attempts to find the next subsequence of the
562         * input sequence that matches the pattern, starting at the specified index.
563         * <p>
564         * If the match succeeds then more information can be obtained via the
565         * <tt>start</tt>, <tt>end</tt>, and <tt>group</tt> methods, and subsequent
566         * invocations of the {@link #find()} method will start at the first
567         * character not matched by this match.
568         * </p>
569         * 
570         * @throws IndexOutOfBoundsException If start is less than zero or if start
571         *         is greater than the length of the input sequence.
572         * @return <tt>true</tt> if, and only if, a subsequence of the input
573         *         sequence starting at the given index matches this matcher's
574         *         pattern
575         */
576        public boolean find(int start) {
577            int limit = getTextLength();
578            if ((start < 0) || (start > limit)) {
579                throw new IndexOutOfBoundsException("Illegal start index");
580            }
581            reset();
582            return search(start);
583        }

585        /**
586         * Attempts to match the input sequence, starting at the beginning of the
587         * region, against the pattern.
588         * <p>
589         * Like the {@link #matches matches} method, this method always starts at
590         * the beginning of the region; unlike that method, it does not require that
591         * the entire region be matched.
592         * <p>
593         * If the match succeeds then more information can be obtained via the
594         * <tt>start</tt>, <tt>end</tt>, and <tt>group</tt> methods.
595         * </p>
596         * 
597         * @return <tt>true</tt> if, and only if, a prefix of the input sequence
598         *         matches this matcher's pattern
599         */
600        public boolean lookingAt() {
601            return match(from, NOANCHOR);
602        }

604        /**
605         * Returns a literal replacement <code>String</code> for the specified
606         * <code>String</code>. This method produces a <code>String</code> that will
607         * work as a literal replacement <code>s</code> in the
608         * <code>appendReplacement</code> method of the {@link Matcher} class. The
609         * <code>String</code> produced will match the sequence of characters in
610         * <code>s</code> treated as a literal sequence. Slashes ('\') and dollar
611         * signs ('$') will be given no special meaning.
612         * 
613         * @param s The string to be literalized
614         * @return A literal string replacement
615         * @since 1.5
616         */
617        public static String quoteReplacement(String s) {
618            if ((s.indexOf('\\') == -1) && (s.indexOf('$') == -1)) {
619                return s;
620            }
621            StringBuffer sb = new StringBuffer();
622            for (int i = 0; i < s.length(); i++) {
623                char c = s.charAt(i);
624                if (c == '\\') {
625                    sb.append('\\');
626                    sb.append('\\');
627                }
628                else if (c == '$') {
629                    sb.append('\\');
630                    sb.append('$');
631                }
632                else {
633                    sb.append(c);
634                }
635            }
636            return sb.toString();
637        }

639        /**
640         * Implements a non-terminal append-and-replace step.
641         * <p>
642         * This method performs the following actions:
643         * </p>
644         * <ol>
645         * <li>
646         * <p>
647         * It reads characters from the input sequence, starting at the append
648         * position, and appends them to the given string buffer. It stops after
649         * reading the last character preceding the previous match, that is, the
650         * character at index {@link #start()}&nbsp;<tt>-</tt>&nbsp;<tt>1</tt>.
651         * </p>
652         * </li>
653         * <li>
654         * <p>
655         * It appends the given replacement string to the string buffer.
656         * </p>
657         * </li>
658         * <li>
659         * <p>
660         * It sets the append position of this matcher to the index of the last
661         * character matched, plus one, that is, to {@link #end()}.
662         * </p>
663         * </li>
664         * </ol>
665         * <p>
666         * The replacement string may contain references to subsequences captured
667         * during the previous match: Each occurrence of <tt>$</tt><i>g</i><tt></tt>
668         * will be replaced by the result of evaluating {@link #group(int) group}
669         * <tt>(</tt><i>g</i><tt>)</tt>. The first number after the <tt>$</tt> is
670         * always treated as part of the group reference. Subsequent numbers are
671         * incorporated into g if they would form a legal group reference. Only the
672         * numerals '0' through '9' are considered as potential components of the
673         * group reference. If the second group matched the string <tt>"foo"</tt>,
674         * for example, then passing the replacement string <tt>"$2bar"</tt> would
675         * cause <tt>"foobar"</tt> to be appended to the string buffer. A dollar
676         * sign (<tt>$</tt>) may be included as a literal in the replacement string
677         * by preceding it with a backslash (<tt>\$</tt>).
678         * <p>
679         * Note that backslashes (<tt>\</tt>) and dollar signs (<tt>$</tt>) in the
680         * replacement string may cause the results to be different than if it were
681         * being treated as a literal replacement string. Dollar signs may be
682         * treated as references to captured subsequences as described above, and
683         * backslashes are used to escape literal characters in the replacement
684         * string.
685         * <p>
686         * This method is intended to be used in a loop together with the
687         * {@link #appendTail appendTail} and {@link #find find} methods. The
688         * following code, for example, writes <tt>one dog two dogs in the
689         * yard</tt> to the standard-output stream:
690         * </p>
691         * <blockquote>
692         * 
693         * <pre>
694         * Pattern p = Pattern.compile(&quot;cat&quot;);
695         * Matcher m = p.matcher(&quot;one cat two cats in the yard&quot;);
696         * StringBuffer sb = new StringBuffer();
697         * while (m.find()) {
698         *     m.appendReplacement(sb, &quot;dog&quot;);
699         * }
700         * m.appendTail(sb);
701         * System.out.println(sb.toString());
702         * </pre>
703         * 
704         * </blockquote>
705         * 
706         * @param sb The target string buffer
707         * @param replacement The replacement string
708         * @return This matcher
709         * @throws IllegalStateException If no match has yet been attempted, or if
710         *         the previous match operation failed
711         * @throws IndexOutOfBoundsException If the replacement string refers to a
712         *         capturing group that does not exist in the pattern
713         */
714        public Matcher appendReplacement(StringBuffer sb, String replacement) {
715    
716            // If no match, return error
717            if (first < 0) {
718                throw new IllegalStateException("No match available");
719            }
720    
721            // Process substitution string to replace group references with groups
722            int cursor = 0;
723            String s = replacement;
724            StringBuffer result = new StringBuffer();
725    
726            while (cursor < replacement.length()) {
727                char nextChar = replacement.charAt(cursor);
728                if (nextChar == '\\') {
729                    cursor++;
730                    nextChar = replacement.charAt(cursor);
731                    result.append(nextChar);
732                    cursor++;
733                }
734                else if (nextChar == '$') {
735                    // Skip past $
736                    cursor++;
737    
738                    // The first number is always a group
739                    int refNum = replacement.charAt(cursor) - '0';
740                    if ((refNum < 0) || (refNum > 9)) {
741                        throw new IllegalArgumentException(
742                                "Illegal group reference");
743                    }
744                    cursor++;
745    
746                    // Capture the largest legal group string
747                    boolean done = false;
748                    while (!done) {
749                        if (cursor >= replacement.length()) {
750                            break;
751                        }
752                        int nextDigit = replacement.charAt(cursor) - '0';
753                        if ((nextDigit < 0) || (nextDigit > 9)) { // not a number
754                            break;
755                        }
756                        int newRefNum = (refNum * 10) + nextDigit;
757                        if (groupCount() < newRefNum) {
758                            done = true;
759                        }
760                        else {
761                            refNum = newRefNum;
762                            cursor++;
763                        }
764                    }
765    
766                    // Append group
767                    if (group(refNum) != null) {
768                        result.append(group(refNum));
769                    }
770                }
771                else {
772                    result.append(nextChar);
773                    cursor++;
774                }
775            }
776    
777            // Append the intervening text
778            sb.append(getSubSequence(lastAppendPosition, first));
779            // Append the match substitution
780            sb.append(result.toString());
781    
782            lastAppendPosition = last;
783            return this;
784        }

805        /**
806         * Replaces every subsequence of the input sequence that matches the pattern
807         * with the given replacement string.
808         * <p>
809         * This method first resets this matcher. It then scans the input sequence
810         * looking for matches of the pattern. Characters that are not part of any
811         * match are appended directly to the result string; each match is replaced
812         * in the result by the replacement string. The replacement string may
813         * contain references to captured subsequences as in the
814         * {@link #appendReplacement appendReplacement} method.
815         * <p>
816         * Note that backslashes (<tt>\</tt>) and dollar signs (<tt>$</tt>) in the
817         * replacement string may cause the results to be different than if it were
818         * being treated as a literal replacement string. Dollar signs may be
819         * treated as references to captured subsequences as described above, and
820         * backslashes are used to escape literal characters in the replacement
821         * string.
822         * <p>
823         * Given the regular expression <tt>a*b</tt>, the input
824         * <tt>"aabfooaabfooabfoob"</tt>, and the replacement string <tt>"-"</tt>,
825         * an invocation of this method on a matcher for that expression would yield
826         * the string <tt>"-foo-foo-foo-"</tt>.
827         * <p>
828         * Invoking this method changes this matcher's state. If the matcher is to
829         * be used in further matching operations then it should first be reset.
830         * </p>
831         * 
832         * @param replacement The replacement string
833         * @return The string constructed by replacing each matching subsequence by
834         *         the replacement string, substituting captured subsequences as
835         *         needed
836         */
837        public String replaceAll(String replacement) {
838            reset();
839            boolean result = find();
840            if (result) {
841                StringBuffer sb = new StringBuffer();
842                do {
843                    appendReplacement(sb, replacement);
844                    result = find();
845                } while (result);
846                appendTail(sb);
847                return sb.toString();
848            }
849            return text.toString();
850        }

1115        /**
1116         * Initiates a search to find a Pattern within the given bounds. The groups
1117         * are filled with default values and the match of the root of the state
1118         * machine is called. The state machine will hold the state of the match as
1119         * it proceeds in this matcher. Matcher.from is not set here, because it is
1120         * the "hard" boundary of the start of the search which anchors will set to.
1121         * The from param is the "soft" boundary of the start of the search, meaning
1122         * that the regex tries to match at that index but ^ won't match there.
1123         * Subsequent calls to the search methods start at a new "soft" boundary
1124         * which is the end of the previous match.
1125         */
1126        boolean search(int from) {
1127            this.hitEnd = false;
1128            this.requireEnd = false;
1129            from = from < 0 ? 0 : from;
1130            this.first = from;
1131            this.oldLast = oldLast < 0 ? from : oldLast;
1132            for (int i = 0; i < groups.length; i++) {
1133                groups[i] = -1;
1134            }
1135            acceptMode = NOANCHOR;
1136            boolean result = parentPattern.root.match(this, from, text);
1137            if (!result) {
1138                this.first = -1;
1139            }
1140            this.oldLast = this.last;
1141            return result;
1142        }
