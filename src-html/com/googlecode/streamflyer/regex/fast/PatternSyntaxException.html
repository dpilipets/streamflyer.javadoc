025    package com.googlecode.streamflyer.regex.fast;
026    
027    import sun.security.action.GetPropertyAction;
028    
029    
030    /**
031     * Unchecked exception thrown to indicate a syntax error in a regular-expression
032     * pattern.
033     * 
034     * @author unascribed
035     * @version 1.17, 10/03/23
036     * @since 1.4
037     * @spec JSR-51
038     */
039    
040    @SuppressWarnings({ "serial", "restriction" })
041    // NEW @SuppressWarnings ADDED BY rwoo
042    public class PatternSyntaxException extends IllegalArgumentException {
043    
044        private final String desc;
045    
046        private final String pattern;
047    
048        private final int index;
049    
050        /**
051         * Constructs a new instance of this class.
052         * 
053         * @param desc A description of the error
054         * @param regex The erroneous pattern
055         * @param index The approximate index in the pattern of the error, or
056         *        <tt>-1</tt> if the index is not known
057         */
058        public PatternSyntaxException(String desc, String regex, int index) {
059            this.desc = desc;
060            this.pattern = regex;
061            this.index = index;
062        }
063    
064        /**
065         * Retrieves the error index.
066         * 
067         * @return The approximate index in the pattern of the error, or <tt>-1</tt>
068         *         if the index is not known
069         */
070        public int getIndex() {
071            return index;
072        }
073    
074        /**
075         * Retrieves the description of the error.
076         * 
077         * @return The description of the error
078         */
079        public String getDescription() {
080            return desc;
081        }
082    
083        /**
084         * Retrieves the erroneous regular-expression pattern.
085         * 
086         * @return The erroneous pattern
087         */
088        public String getPattern() {
089            return pattern;
090        }
091    
092        private static final String nl = java.security.AccessController
093                .doPrivileged(new GetPropertyAction("line.separator"));
094    
095        /**
096         * Returns a multi-line string containing the description of the syntax
097         * error and its index, the erroneous regular-expression pattern, and a
098         * visual indication of the error index within the pattern.
099         * 
100         * @return The full detail message
101         */
102        @Override
103        public String getMessage() {
104            StringBuffer sb = new StringBuffer();
105            sb.append(desc);
106            if (index >= 0) {
107                sb.append(" near index ");
108                sb.append(index);
109            }
110            sb.append(nl);
111            sb.append(pattern);
112            if (index >= 0) {
113                sb.append(nl);
114                for (int i = 0; i < index; i++) {
115                    sb.append(' ');
116                }
117                sb.append('^');
118            }
119            return sb.toString();
120        }
121    
122    }
